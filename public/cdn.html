CDN 资源配置

此文件包含用于生产环境的 CDN 资源链接

将以下配置添加到 nuxt.config.ts 中：

// 生产环境使用 CDN
const isProduction = process.env.NODE_ENV === 'production'

export default defineNuxtConfig({
  vite: {
    build: {
      rollupOptions: {
        external: isProduction ? [
          'three',
          'echarts',
          'chart.js'
        ] : []
      }
    }
  }
})
